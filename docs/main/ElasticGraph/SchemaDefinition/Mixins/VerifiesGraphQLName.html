def self.verify_name!(name)
  return if GRAPHQL_NAME_PATTERN.match?(name)
  raise Errors::InvalidGraphQLNameError, "Not a valid GraphQL name: `#{name}`. #{GRAPHQL_NAME_VALIDITY_DESCRIPTION}"
end
